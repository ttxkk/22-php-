git init ：初始化git仓库
rm -rf .git ：移除仓库
git add 文件名 ：将文件添加到暂存区
git commit : 将文件添加到仓库
git commit -m "备注内容" ：提交修改内容并备注
git checkout -b [名称] ：添加一个分支，分支名为名称
git branch ：检查有什么分支
git checkout 分支名 ：切换分支
git merge [分支名] ： 合并分支到主分支
git log ：查看修改日志
git log --stat ：查看具体修改内容
git diff [commit id]：查看具体commitId修改的内容
git reset --hard [commit id] / git checkout [commit id] ：代码回溯
